Y(t+\varDelta t)&=&Y(t)+DY(t)\varDelta t+B(t)\sqrt{\frac{\hbar}{m}}\sqrt{\varDelta t} \\
Z(t+\varDelta t)&=&Z(t)+DZ(t)\varDelta t+C(t)\sqrt{\frac{\hbar}{m}}\sqrt{\varDelta t}
